## CIVIL WAR ##

GSR_civil_war = {
    #Fun fact! Having more divs are better then having no divs
	allowed = {
		original_tag = GSR
	}
	enable = {
		country_exists = GPG
		country_exists = GDR 
		country_exists = GFG
	}

	abort = {
		has_war = no
	}

	ai_strategy = {
		type = production_upgrade_desire_offset
		id = infantry_equipment_1
		value = -1000
	}
	ai_strategy = {
		type = equipment_production_factor
		id = infantry
		value = 65
	}
	ai_strategy = {
		type = ignore_army_incompetence
		value = 50
	}
	ai_strategy = {
		type = force_build_armies
		value = 30
	}
}

GSR_militarize = {
    #Tells GSR to build mils as they have many civs
	allowed = {
		original_tag = GSR
	}
	enable = {
		industrial_complex > 10
		arms_factory < 10
	}

	abort = {
		arms_factory > 10
	}

	ai_strategy = {
		type = building_target
		id = arms_factory
		value = 300
	}
}

GSR_huge_army = {
    #Once GSR is at peace to have a large army ready
	allowed = {
		original_tag = GSR
	}
	enable = {
		has_war = no
		has_army_size = {
            size < 30
        }
	}

	abort = {
		has_army_size = {
            size > 30
        }
	}

	ai_strategy = {
		type = build_army
		id = infantry
		value = 250
	}
	ai_strategy = {
		type = ignore_army_incompetence
		value = 25
	}
}
