## ARMY PLACEMENT ##

WGF_GMD_preparation = {
    #WGF sends troops to GMD
	allowed = {
		original_tag = WGF
	}
	enable = {
		OR = {
			focus_progress = {
				focus = WGF_focus_flip_the_coin
				progress > 0.01
			}
			AND = {
				country_exists = GMD
				has_completed_focus = WGF_focus_flip_the_coin
			}
		}
	}

	abort = {
		NOT = { country_exists = GMD }
	}

	ai_strategy = {
        type = prepare_for_war
        id = GMD
        value = 250
    }
}

WGF_GMD_war = {
    #Tell Firey that 10 divs can shockingly beat 1 div (NO WAY!!!!!!!!!!!!)
	allowed = {
		original_tag = WGF
	}
	enable = {
		has_war_with = GMD
	}

	abort = {
		NOT = { country_exists = GMD }
	}

	ai_strategy = {
		type = front_unit_request
		tag = GMD
		value = 500
	}
}

WGF_militarize = {
    #Tells WGF to build mils early on to counter Coiny
	allowed = {
		original_tag = WGF
	}
	enable = {
		arms_factory < 6
		country_exists = GMD
	}

	abort = {
		OR = {
			NOT = {country_exists = GMD}
			arms_factory > 6
		}	
	}

	ai_strategy = {
		type = building_target
		id = arms_factory
		value = 300
	}
}

WGF_GSR_aid = {
    #Tell Firey if GSR is in civil war aid them
	allowed = {
		original_tag = WGF
	}
	enable = {
		has_war = no
		country_exists = GSR
		OR = {
			GSR = {has_war_with = GDR}
			GSR = {has_war_with = GPG}
			GSR = {has_war_with = GFG}
		}
	}

	abort = {
		OR = {
			has_war = yes
			GSR = {has_war = no}
			NOT = {country_exists = GSR}
		}
	}


	ai_strategy = {
		type = equipment_production_factor
		id = infantry
		value = 50
	}
	ai_strategy = {
		type = send_lend_lease_desire
		id = GSR
		value = 50
	}
