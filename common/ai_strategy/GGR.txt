## ARMY PLACEMENT ##

GGR_protect_raiders = {
    #Tells GGR to defend from BUN till anarchists are done
    allowed = {
		original_tag = GGR
	}
	enable = {
		has_army_size = {
            size < 7
        }
	}

	abort = {
		OR = {
			has_army_size = {
                size > 7
            }
            NOT = {country_exists = BUN}
		}
	}

	ai_strategy = {
        type = prepare_for_war
        id = BUN
        value = 200
    }
}


GGR_FFT_preparation_and_war = {
    #GGR sends troops to FFT
	allowed = {
		original_tag = GGR
	}
	enable = {
		OR = {
			focus_progress = {
				focus = GGR_focus_partizan_war
				progress > 0.01
			}
			has_war_with = FFT
		}
	}

	abort = {
		NOT = { country_exists = FFT }
	}

	ai_strategy = {
        type = prepare_for_war
        id = FFT
        value = 500
    }
	ai_strategy = {
		type = garrison
		value = -85
	}
}

GGR_FFT_spy = {
    #GGR will spy on FFT to gain more intel
	allowed = {
		original_tag = GGR
	}
	enable = {
		country_exists = FFT
	}

	abort = {
		NOT = { country_exists = FFT }
	}

	ai_strategy = {
		type = operative_mission
		value = 50
		state = 169
		mission = build_intel_network
		mission_target = FFT
	}
}

## INDUSTRY ##

GGR_tanks = {
    #Once GGR has units and tank tech bonus it will try to make tanks for the rest of the game
	allowed = {
		original_tag = GGR
	}
	enable = {
		NOT = {has_dlc = "No Step Back"}
		has_army_size = {
            size > 8
        }
		has_completed_focus = GGR_focus_forest_tanks
	}

	abort = {
		has_army_size = {
			size > 1
			type = armor
		}
	}


	ai_strategy = {
		type = build_army
		id = armor
		value = 1000
	}
	ai_strategy = {
		type = equipment_production_factor
		id = armor
		value = 75
	}
	ai_strategy = {
		type = research_weight_factor
		id = gwtank
		value = 1000
	}
	ai_strategy = {
		type = research_weight_factor
		id = basic_light_tank
		value = 750
	}
	ai_strategy = {
		type = research_weight_factor
		id = tech_trucks
		value = 100
	}
}
